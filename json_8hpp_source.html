49template<typename Base>
50std::unique_ptr<Base> Create(const nlohmann::json & json)
51{
52 if (!json.is_object() || json.size() != 1) {
53 throw std::logic_error("Expected dictionary of size 1 of format {tag: object}");
54 }
55 return gInstance<Factory<Base, const nlohmann::json &>>().create(json.begin().key(), json.begin().value());
56}
