99template<typename... Ts>
100class variant_tags<std::variant<Ts...>>
101{
102public:
103 template<typename T>
104 requires(std::is_same_v<T, Ts> || ...)
105 static void Add(const std::string & tag)
106 {
107 const auto idx = ezconfig::index_in_typepack<T, Ts...>();
108 s_tag_to_idx[tag] = idx;
109 s_idx_to_tag[idx] = tag;
110 }
111
112 static std::variant<Ts...> TagToType(const std::string & tag)
113 {
114 auto idx = s_tag_to_idx.at(tag);
115 if (idx == 0) { return std::variant<Ts...>{std::in_place_index<0>}; }
116 if (idx == 1) { return std::variant<Ts...>{std::in_place_index<1>}; }
117 throw std::runtime_error("WIP");
118 }
119
120 template<typename T>
121 static std::string TypeToTag()
122 {
123 const auto idx = ezconfig::index_in_typepack<T, Ts...>();
124 return s_idx_to_tag.at(idx);
125 }
126
127private:
128 static std::map<std::string, std::size_t> s_tag_to_idx;
129 static std::map<std::size_t, std::string> s_idx_to_tag;
130};
