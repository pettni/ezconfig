33 template<typename Derived, typename Intermediate = Derived>
34 requires(std::is_base_of_v<Base, Derived>)
35 static void Register(const std::string & tag)
36 {
37 auto creator = [](JsonArg json) {
38 auto interm = json.get<Intermediate>();
39 return std::make_unique<Derived>(std::move(interm));
40 };
41 FactoryBase::Register(tag, std::move(creator));
42 }

47 static std::unique_ptr<Base> Create(JsonArg json)
48 {
49 if (!json.is_object() || json.size() != 1) {
50 throw std::logic_error("Expected dictionary of size 1 of format {tag: object}");
51 }
52 return FactoryBase::Create(json.begin().key(), json.begin().value());
53 }
